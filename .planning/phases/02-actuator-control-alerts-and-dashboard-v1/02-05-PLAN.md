**hub/dashboard/src/lib/SensorChart.svelte** — create per UI-SPEC and Research Pattern 6:

Props: `zoneId: string`, `sensorType: string`, `days: number` (bindable, default 7).

Implementation details:
- Import `uPlot` from `'uplot'` and `'uplot/dist/uPlot.min.css'`
- Use `onMount` to call `loadAndRender()` and `onDestroy` to call `chart?.destroy()`
- Use `$effect(() => { if (days) loadAndRender(); })` to re-render on range toggle
- Fetch from `/api/zones/${zoneId}/history?sensor_type=${sensorType}&days=${days}`
- Transform response: `timestamps = data.map(d => new Date(d.ts).getTime() / 1000)`, `values = data.map(d => d.value)`
- Use `requestAnimationFrame` inside `onMount` before reading `container.clientWidth` (Pitfall 2 from Research)
- Attach `ResizeObserver` to container, call `chart.setSize({ width: container.clientWidth, height })` on resize (anti-pattern from Research)
- Series colors per sensor type: moisture `#4ade80`, ph `#60a5fa`, temperature `#fb923c`
- uPlot config: grid stroke `#2d3149`, axis stroke `#94a3b8`, axis tick size `14` (not 12 — Typography canvas exception), cursor stroke `#6b7280`, series width 1.5
- Height: 160px on mobile, 200px on desktop (media query `min-width: 640px`)
- Loading state: skeleton div with `--color-surface` background, same dimensions
- Empty state: centered text "No data for this period" in `--color-muted`

The time range toggle (7 days / 30 days) is two buttons above the chart. Active: `--color-accent` text + underline. Inactive: `--color-muted`. Toggle switches `days` prop, which triggers `$effect` re-fetch.

Section heading "Sensor History" (Heading role, 20px/600) above the three charts on the zone detail page (not inside SensorChart — rendered by the page).

<script lang="ts">
  import { page } from '$app/stores';
  import { dashboardStore } from '$lib/ws.svelte';
  import SensorValue from '$lib/SensorValue.svelte';
  import HealthBadge from '$lib/HealthBadge.svelte';
  import CommandButton from '$lib/CommandButton.svelte';
  import SensorChart from '$lib/SensorChart.svelte';
  import type { HealthScore } from '$lib/types';

  const zoneId = $derived($page.params.id);
  const zone = $derived(dashboardStore.zones.get(zoneId));
  const healthData = $derived(dashboardStore.zoneHealthScores.get(zoneId));
  const healthScore = $derived<HealthScore>(healthData?.score ?? 'green');
  const contributingSensors = $derived(healthData?.contributing_sensors ?? []);
  const irrigationState = $derived(
    dashboardStore.actuatorStates.get(`irrigation:${zoneId}`) ?? 'closed'
  );

  let irrigateLoading = $state(false);
  let chartDays = $state(7);

  async function sendIrrigationCommand(action: 'open' | 'close') {
    irrigateLoading = true;
    try {
      const res = await fetch('/api/actuators/irrigate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ zone_id: zoneId, action }),
      });
      if (!res.ok) {
        const data = await res.json().catch(() => ({}));
        const msg = data.detail || 'Command failed \u2014 tap to retry';
        window.dispatchEvent(new CustomEvent('farm:toast', { detail: { message: msg } }));
      }
    } catch {
      window.dispatchEvent(new CustomEvent('farm:toast', {
        detail: { message: 'Command failed \u2014 tap to retry' },
      }));
    } finally {
      irrigateLoading = false;
    }
  }
</script>

<h2 class="page-heading">{zoneId}</h2>

{#if zone}
  <section class="irrigation-section">

    </div>

    {/if}
    <div class="irrigation-controls">
      <CommandButton
        label="Open valve"
        loading={irrigateLoading && irrigationState !== 'open'}
        disabled={irrigationState === 'open' || irrigationState === 'moving'}
        onclick={() => sendIrrigationCommand('open')}
      />
      <CommandButton
        label="Close valve"

  </section>

    {/if}
  </section>

  <section class="charts-section">
    <h3 class="section-heading">Sensor History</h3>

  </section>
{:else}

{/if}
