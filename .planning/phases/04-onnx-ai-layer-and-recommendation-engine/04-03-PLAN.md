       - Uses APScheduler AsyncIOScheduler (pinned to 3.x per 04-RESEARCH.md Pitfall 1):
         ```python
         from apscheduler.schedulers.asyncio import AsyncIOScheduler

         ```
       - def start(): Adds three interval jobs:
         * run_zone_health_inference: interval minutes=15, id="zone_health"
         * run_irrigation_inference: interval hours=1, id="irrigation"
         * run_flock_anomaly_inference: interval minutes=30, id="flock_anomaly"
         Then calls self._scheduler.start(). Per 04-RESEARCH.md Pattern 4: start() must be called inside async context.

       - async def run_zone_health_inference():
         For each zone in zone_config_store:
           1. Check ai_settings.get_mode("zone_health") -- if "rules", skip
           2. Check InferenceService("zone_health").is_loaded -- if not, skip
           3. Check feature_aggregator.check_data_maturity(zone_id) -- if gate not passed, skip
           4. features = await feature_aggregator.aggregate_zone_features(zone_id, ["moisture", "ph", "temperature"], window_hours=24)
           5. If features is None (insufficient data), skip
           6. vector = feature_aggregator.build_feature_vector(features, ["moisture", "ph", "temperature"])
           7. result = inference_services["zone_health"].infer(vector)
           8. If result is None (low confidence), skip -- rule engine handles it
           9. health_dict = inference_services["zone_health"].format_zone_health_result(zone_id, result[0], result[1], ["moisture", "ph", "temperature"])
           10. await notify_callback(health_dict)

       - async def run_irrigation_inference():
         Similar pattern for irrigation domain. If AI mode and model loaded and gate passed:
           1. Aggregate features for each zone
           2. Infer
           3. If prediction=1 (irrigate needed): format_recommendation(zone_id, "irrigate", ...)
           4. maturity_tracker.record_recommendation("irrigation", rec_id)
           5. await notify_callback({"type": "recommendation_queue", "recommendations": [rec_dict]})

       - async def run_flock_anomaly_inference():
         Similar for flock. Uses flock-specific sensor_types=["feed_weight", "nesting_box_weight", "water_level"] and window_hours=168 (7 days).

       - async def trigger_zone_reinference(zone_id: str):
         Called on threshold crossing. Runs zone_health + irrigation inference for just that zone immediately. Skips if mode is "rules".

       - def stop(): self._scheduler.shutdown(wait=False)

       import logging, os, aiohttp
       from fastapi import APIRouter, HTTPException
       from pydantic import BaseModel

       router = APIRouter()
       BRIDGE_INTERNAL_URL = os.getenv("BRIDGE_INTERNAL_URL", "http://bridge:8001")

       class AISettingsPatch(BaseModel):
           domain: str  # "irrigation" | "zone_health" | "flock_anomaly"
           mode: str    # "ai" | "rules"

       @router.get("/api/settings/ai")
       async def get_ai_settings():
           url = f"{BRIDGE_INTERNAL_URL}/internal/ai-settings"
           async with aiohttp.ClientSession() as session:
               async with session.get(url, timeout=aiohttp.ClientTimeout(total=5)) as resp:
                   if resp.status != 200:
                       raise HTTPException(status_code=resp.status, detail="Bridge error")
                   return await resp.json()

       @router.patch("/api/settings/ai")
       async def patch_ai_settings(body: AISettingsPatch):
           url = f"{BRIDGE_INTERNAL_URL}/internal/ai-settings"
           async with aiohttp.ClientSession() as session:
