       - import threading, os, logging, uuid, onnxruntime as ort, numpy as np
       - MIN_CONFIDENCE = float(os.getenv("MIN_CONFIDENCE", "0.65"))
       - MODELS_DIR = os.getenv("MODELS_DIR", os.path.join(os.path.dirname(__file__), "..", "..", "models"))
       - Constructor takes model_name: str (e.g. "zone_health"), loads from MODELS_DIR/{model_name}.onnx if exists
       - self._lock = threading.Lock()
       - self._session: ort.InferenceSession | None = None
       - self._model_path: str | None = None

       - def _load(self, model_path: str):
         Creates InferenceSession with CPUExecutionProvider. Stores input_name and output_names.
         Under self._lock, swaps self._session reference.

       - def reload(self, model_path: str):
         Called from watchdog handler (different thread). Calls self._load(model_path).
         Logs info: "Reloaded model from {model_path}"

       - def infer(self, feature_vector: list[float]) -> tuple[float, float] | None:
         Acquires self._lock to read session reference, then releases before running inference.
         If session is None, return None.
         Converts feature_vector to np.array([feature_vector], dtype=np.float32).
         Runs session.run(None, {input_name: arr}).
         For classification models: result[1] contains class probabilities. confidence = max probability.
         If confidence < MIN_CONFIDENCE, return None (fall back to rules per D-04).
         Returns (predicted_class_index, confidence).

       - def format_zone_health_result(self, zone_id: str, prediction: int, confidence: float, feature_names: list[str]) -> dict:
         Maps prediction index to score: {0: "green", 1: "yellow", 2: "red"}.
         Returns {"type": "zone_health_score", "zone_id": zone_id, "score": score_label, "contributing_sensors": [names of sensors with extreme feature values], "source": "ai"}.

       - def format_recommendation(self, zone_id: str, rec_type: str, prediction: int, confidence: float, sensor_summary: str) -> dict | None:
         If prediction indicates no action needed, return None.
         Returns recommendation dict with same keys as RuleEngine output PLUS "source": "ai":
         {"recommendation_id": str(uuid.uuid4()), "zone_id": zone_id, "rec_type": rec_type, "action_description": f"AI: {description}", "sensor_reading": sensor_summary, "explanation": f"AI model confidence: {confidence:.0%}", "source": "ai"}

       - @property
       - def is_loaded(self) -> bool: return self._session is not None

       - Purpose: Train a classifier that predicts zone health score (green=0, yellow=1, red=2) from aggregated sensor features.
       - Input features (12): moisture [mean, min, max, std], ph [mean, min, max, std], temperature [mean, min, max, std]

       - async def fetch_training_data(db_pool) -> tuple[np.ndarray, np.ndarray]:
         Queries sensor_readings with quality='GOOD' grouped by zone_id and time_bucket('1 hour', time).
         For each time bucket, computes aggregate features (same as FeatureAggregator).
         Labels each bucket using the threshold logic from health_score.py (green/yellow/red based on ranges).
         Returns (X: np.ndarray shape [n_samples, 12], y: np.ndarray shape [n_samples] with values 0/1/2).
       - def train_and_export(X, y, output_path: str, prev_model_path: str | None = None) -> bool:
         Trains GradientBoostingClassifier(n_estimators=100, max_depth=5, random_state=42).
         80/20 train/test split with stratify=y.
         If prev_model_path exists: load previous model, evaluate on test set. If new model F1-macro <= previous F1-macro, log warning "Regression detected — keeping previous model" and return False.
         Export to ONNX: convert_sklearn(clf, initial_types=[("float_input", FloatTensorType([None, 12]))], target_opset=12).
         Validate with onnx.checker.check_model().
         Save to output_path. If prev model existed and new model is better, rename prev to {output_path}.prev.onnx (keep last 2 versions per D-08).
         Return True if new model was saved.
       - async def main(): entry point, fetches data, trains, exports to hub/models/zone_health.onnx

|----------|-------------|

| training data -> model | Poisoned data could produce bad predictions |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-04 | Tampering | .onnx model files | mitigate | onnx.checker.check_model() validates every model before hot-reload; models/ directory only writable by bridge process |
| T-04-05 | Tampering | training data | mitigate | GOOD-flag filter in training SQL excludes tampered readings already flagged as SUSPECT/BAD |
| T-04-06 | Denial of Service | InferenceService.infer() | accept | ONNX inference for tree models is < 5ms; fixed input shape enforced by InferenceSession |
| T-04-07 | Spoofing | model hot-reload | mitigate | watchdog only monitors specific directory; onnx.checker validates before session swap |
</threat_model>
